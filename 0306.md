#### call()
``` javascript
Function.prototype.call(context = window, ...args) {
    let fn = Symbol('fn');
    context[fn] = this;
    let res = context[fn](args);
    delete context[fn];
    return res;
}
```

#### bind()
``` javascript
Function.prototype.bind(context = window, ...args) {
    let _this = this;

    function dBound() {
        return _this.apply(this instanceof dBound ? this : context, args);
    }
    dBound.prototype = this.prototype;
    return dBound();
}
```
