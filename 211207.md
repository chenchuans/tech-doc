# 设计模式之单例模式

> 又称作包装器模式，讲一个一个接口转换为用户使用的另一个接口，解决类（对象）的不兼容问题。旧接口和使用者不匹配，中间加一个适配器转换接口

## es6 写法

缺点： 告诉使用者必须用 `getInstance` 方法才能调用

``` typescript
class Single {
    private static instance: Single;
    private constructor() {

    }
    public static getInstance() {
        if (!Single.instance) {
            Single.instance = new Single();
        }
        return Single.instance;
    }
}
const s1 = Single.getInstance();
const s2 = Single.getInstance();

console.log(s1 === s2); // true
```

## es5 写法

``` typescript
function Single() {

}

Single.getInstance = (function() {
    let instance: Single;
    if (!instance) {
        instance = new Single();
    }
    return instance;
})()

const s1 = Single.getInstance();
const s2 = Single.getInstance();

console.log(s1 === s2); // true
```

## 单例与构建过程的分离写法

``` typescript
/**
 * 单例与构建过程的分离
 *  */
export { }
function Window() { }
interface Window {
    new(): Window;
}

// 此方法构建任何方法
let createInstance = function (Constructor: any) {
    let instance: any;
    return function (this: any) {
        if (!instance) {
            Constructor.apply(this, arguments)
            // this.__proto__ = Constructor.prototype  修正原型指向
            Object.setPrototypeOf(this, Constructor.prototype)
            instance = this
        }
        return this;
    }
}

let createWindow = createInstance(Window)
let c1 = createWindow()
let c2 = createWindow()
console.log(c1 === c2);
```

## 在 `webpack` require函数中的使用

``` typescript
(function (modules) {
    // webpack启动函数
    // 定义模块的缓存
    var installedModules: Record<string, any> = {};
    // 定义在浏览器中使用的require方法
    function _webpack_require_(moduleId: string) {
        // 检查是否在浏览器缓存中
        if (installedModules[moduleId]) {
            return installedModules[moduleId].exports;
        }

        // 创建模块放到浏览器缓存中
        var module = (installedModules[moduleId] = {
            i: moduleId,
            l: false,
            exports: {},
        })

        // 执行模块函数
        module[moduleId].call(
            module.exports,
            module,
            module.exports,
            _webpack_require_,
        )

        // 模块加载过后为 true
        module.l = true

        // 返回模块加载内容
        return module.exports
    }
})({
    './src/index.js': function (module: any, exports: any) {
    }
})
```

## 在模态框中使用

``` html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>模态框使用</title>
</head>
<body>
    <button id="show-modal">显示模态框</button>
    <button id="hide-modal">隐藏模态框</button>
    <script>
        class LoginModal {
            constructor() {
                this._element = document.createElement('div');
                this._element.innerHTML = (`
                    <form>
                        用户名<input/>
                        <button>登录</button>
                    </form>
                `)
                this._element.style.cssText = 'width: 200px; height: 200px; background: red; position: absolute; top: 50%; left: 50%;'
                document.body.appendChild(this._element)
            }
            show() {
                this._element.style.display = 'block'
            }
            hide() {
                this._element.style.display = 'none'
            }
            static getInstance() {
                if (!LoginModal.instance) {
                    LoginModal.instance = new LoginModal()
                }
                return LoginModal.instance
            }
        }
        document.getElementById('show-modal').addEventListener('click', () => {
            LoginModal.getInstance().show()
        });
        document.getElementById('hide-modal').addEventListener('click', () => {
            LoginModal.getInstance().hide()
        });
    </script>
</body>
</html>
```

## express 简单的读取缓存中

``` typescript
const express = require('express')
const app = express()
const fs = require('fs')

// 缓存放到外面定义，一定要为单例模式，不管谁用都可以拿到
let cache: Record<string, any> = {}
app.get('api/user/:id', function(res, req) {
    const id = req.params.id
    if (cache[id]) {
        res.json(cache[id])
    } else {
        fs.redFile(`./${id}.json`, 'utf-8', (err, data) => {
            const info = JSON.parse(data)
            cache[id] = info
            res.json(info)
        })
    }
})

app.listen(8080)
```
